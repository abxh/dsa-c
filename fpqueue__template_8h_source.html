462static inline void JOIN(internal, JOIN(FPQUEUE_NAME, upheap))(FPQUEUE_TYPE *self, uint32_t index)
463{
464 assert(self != NULL);
465 assert(index < self->count);
466
467 uint32_t parent;
468 while (index > 0) {
469 parent = FPQUEUE_PARENT(index);
470
471 const bool sorted = self->elements[parent].priority >= self->elements[index].priority;
472
473 if (sorted) {
474 break;
475 }
476
477 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
478 self->elements[index] = self->elements[parent];
479 self->elements[parent] = temp;
480
481 index = parent;
482 }
483}
484
485static inline void JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(FPQUEUE_TYPE *self, const uint32_t index)
486{
487 assert(self != NULL);
488 assert(self->count == 0 || index < self->count);
489
490 const uint32_t l = FPQUEUE_LEFT_CHILD(index);
491 const uint32_t r = FPQUEUE_RIGHT_CHILD(index);
492
493 uint32_t largest = index;
494 if (l < self->count && self->elements[l].priority > self->elements[index].priority) {
495 largest = l;
496 }
497 if (r < self->count && self->elements[r].priority > self->elements[largest].priority) {
498 largest = r;
499 }
500
501 if (largest == index) {
502 return;
503 }
504
505 const FPQUEUE_ELEMENT_TYPE temp = self->elements[index];
506 self->elements[index] = self->elements[largest];
507 self->elements[largest] = temp;
508
509 JOIN(internal, JOIN(FPQUEUE_NAME, downheap))(self, largest);
510}
